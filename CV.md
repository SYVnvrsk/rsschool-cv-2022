function toCamelCase(str){
  let answer = '';
  if (str != '') {
    let regex = /-|_/g;
    let words = str.split(regex);
    answer += words[0]
    for (let i = 1; i < words.length; i++) {
      answer += words[i].charAt(0).toUpperCase();
      answer += words[i].slice(1);
    }
  }
  return answer;
}
